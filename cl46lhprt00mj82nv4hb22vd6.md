#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
using namespace std;

int main()
{
    int n,arr1[16],arr2[16];
    cin>>n;
    int i,j;
    for(i=0;i<n;i++)
    {
        cin>>arr1[i]>>arr2[i];
    }
    int m=(1<<n)-1;
    for(i=0;i<m;i++)
    {
        bool sign=1;
        for(j=0;j<n;j++)
        {
            if(((i>>(n-1-arr1[j]))&1)!=(((i>>(n-1-j))&1)^1^arr2[j]))
            {
                sign=0;
                break;
            }
        }
        if(sign)
        {
            for(j=n-1;j>=0;j--)
            {
                // cout<<(i>>j)&1>>" ";
                printf("%d ",(i>>j)&1);
            }
            break;
        }
    }
}
